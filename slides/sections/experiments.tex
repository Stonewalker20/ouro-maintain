\section{Experiments}

\begin{frame}{Experimental Questions}
  \begin{enumerate}
    \item Does adaptive depth improve predictive-maintenance performance over fixed-depth baselines?
    \item Do hard windows use more loop steps than easy windows?
    \item Can adaptive looping match stronger baselines at similar latency?
    \item Does early exit reduce average compute without hurting critical-failure recall?
  \end{enumerate}
\end{frame}

\begin{frame}{Evaluation Plan}
  \begin{itemize}
    \item Primary dataset: C-MAPSS FD001 with official test labels.
    \item Models: direct baseline, fixed-depth loop, adaptive loop.
    \item Metrics: accuracy, macro F1, weighted F1, average loop depth.
    \item Secondary path: IMS run-to-failure preprocessing and temporal evaluation.
  \end{itemize}
\end{frame}
