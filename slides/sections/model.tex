  \begin{itemize}
    \item Shared recurrent block refines a latent state for up to $K$ steps.
    \item Learned exit gate stops early on easy examples.
    \item Final heads predict health class, severity, and action.
  \end{itemize}
\end{frame}

\begin{frame}{Training Configuration}
  \begin{itemize}
    \item Baseline: direct classifier on flattened or pooled window features.
    \item Fixed-depth loop: always run $K$ latent refinement steps.
    \item Adaptive loop: stop when exit confidence exceeds threshold.
    \item Encoder options: MLP, GRU, or 1D CNN.
  \end{itemize}
\end{frame}
